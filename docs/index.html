x402 Payment Protocol

Pay-per-request with Solana USDC. No API keys, no subscriptions, no accounts. 

# Autonomous agent pays for each API call via x402
from acedatacloud import AceDataCloud
from acedatacloud_x402 import create_x402_payment_handler, SolanaKeypairSigner

signer = SolanaKeypairSigner.from_base58("your-solana-key")
handler = create_x402_payment_handler(
    network="solana",
    solana_signer=signer,
)

client = AceDataCloud(payment_handler=handler)

# Every call automatically pays via x402
result = client.search.create(query="Solana DeFi trends")    # $0.001 USDC
analysis = client.chat.completions.create(...)               # $0.002 USDC
image = client.images.generate(...)                           # $0.005 USDC
